#include <time.h>

int main(void) {
	clock_t t;

    t = clock();
	// some  code
	t = clock() - t;

    // display time
    double time_taken = ((double)t)/CLOCKS_PER_SEC;

	printf("\n sortowanie scalieniem trwalo %f seconds \n", time_taken);
	return 0;
}
